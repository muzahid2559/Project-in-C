int nob = 0, nor = 0, noc = 0, amount = 0, count = 0;
void main(){
    while(1){
    clrscr();

    case 1:
        Bus();
        break;
    case 2:
        Riksha();

    case 3:
        Cycle();

    case 4:
        ShowDetails();

}
